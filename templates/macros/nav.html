{# Shared page navigation — prev/next links for book-like reading flow. Works for all page types (poem, interlude) in both flat and grouped collections. For interludes in grouped collections, page.higher/lower links to sibling interludes rather than subsections. We override: opening interludes go to the first subsection, closing interludes come from the last subsection. #} {% macro page_nav(page, config) %} {% set current = get_section(path=page.ancestors | last) %} {% set is_interlude = page.extra.entry_type | default(value="") == "interlude" %} {% set grouped = current.subsections | length > 0 %}
{% if is_interlude and grouped and page.extra.position | default(value="start") == "end" %} {% set last_sub = get_section(path=current.subsections | last) %} {% if last_sub.pages | length > 0 %} ← {{ last_sub.pages | last | get(key='title') }} {% else %} ← {{ last_sub.title }} {% endif %} {% elif is_interlude and grouped %} ← {{ current.title }} {% set first_sub = get_section(path=current.subsections | first) %} {{ first_sub.title }} → {% else %} {% if page.lower %} ← {{ page.lower.title }} {% else %} ← {{ current.title }} {% endif %} {% if page.higher %} {{ page.higher.title }} → {% else %} {% if current.ancestors | length > 2 %} {% set up = get_section(path=current.ancestors | last) %} {% set_global next = false %} {% set_global found = false %} {% for p in up.subsections %} {% set sib = get_section(path=p) %} {% if found and not next %} {% set_global next = sib %} {% endif %} {% if sib.path == current.path %} {% set_global found = true %} {% endif %} {% endfor %} {% if next %} {{ next.title }} → {% else %} {# Last subsection — check for closing interlude #} {% set col = get_section(path=current.ancestors | last) %} {% set_global closing = false %} {% for p in col.pages %} {% if p.extra.entry_type | default(value="") == "interlude" and p.extra.position | default(value="start") == "end" %} {% set_global closing = p %} {% endif %} {% endfor %} {% if closing %} {{ closing.title }} → {% else %} {% endif %} {% endif %} {% elif current.subsections | length > 0 %} {% set first = get_section(path=current.subsections | first) %} {{ first.title }} → {% else %} {% endif %} {% endif %} {% endif %}
